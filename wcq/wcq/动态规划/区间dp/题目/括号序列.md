	a = " " + a;
	// memset(dp, -1, sizeof dp);
	rep(i, 1, n - 1) {
		rep(j, 1, n - i) {
			dp[j][j + i] = max(dp[j][j + i], max(dp[j + 1][j + i], dp[j][j + i - 1]));

				dp[j][j + i] = max(dp[j][j + i], dp[j + 1][j + i - 1] + 2);
			rep(k, j, j + i - 1) {
				dp[j][j + i] = max(dp[j][j + i], dp[j][k] + dp[k + 1][j + i]);
			}
		}
	}
