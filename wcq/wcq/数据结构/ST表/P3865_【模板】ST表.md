#include <bits/stdc++.h>
using namespace std;
#define rep(i, a, b) for(int i = a; i <= b; ++i)
#define per(i, a, b) for(int i = a; i >= b; --i)
typedef long long ll;
const int N = 1e5 + 7;


int n, m;
int a[N], f[21][N];

int rmq(int l, int r) {
	int len = __lg(r - l + 1);
	return max(f[len][l], f[len][r - (1 << len) + 1]);
}
void solve() {
	cin >> n >> m;
	for (int i = 1; i <= n; i++) {
		scanf("%d", &a[i]);
		f[0][i] = a[i];
	}
	// ½¨±í
	for (int j = 1; j <= 20; j++) {
		for (int i = 1; i + (1 << j) - 1 <= n; i ++) {
			f[j][i] = max(f[j - 1][i], f[j - 1][i + (1 << (j - 1))]);
		}
	}
	// ²éÑ¯
	while (m--) {
		int l, r;
		scanf("%d %d", &l, &r);
		cout << rmq(l, r) << "\n";
	}
}

int main() {

	solve();

	return 0;
}
