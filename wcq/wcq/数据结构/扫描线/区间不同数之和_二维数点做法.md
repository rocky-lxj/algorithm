#include<bits/stdc++.h>
using namespace std;
#define y1 abbbbb
using ll = long long;
const int N = 2e5 + 7;
int lowbit(int x) {
	return x & (-x);
}

// 注意：这个只是统计个数，不是和
// 区间不同数之和  扫描线 + 树状数组 + 离散化
// 对 x 离散化
// 平行于 x轴 的扫描线，则需把同一水平线上所有的操作事件完成，才能上移扫描线
// 在线 -> 离线
// OJ会卡输入输出的时间

vector<int> vx;
vector<array<int, 4>> event;
int n, q, m;
int bit[N * 8], ans[N], perx[N];
void change(int x, int d) {
	for (; x <= m; x += lowbit(x))
		bit[x] += d;
}
int query(int x) {
	int s = 0;
	for (; x; x -= lowbit(x))
		s += bit[x];
	return s;
}

int main() {
	scanf("%d%d", &n, &q);
	for (int i = 1; i <= n; i++) {
		int x;
		cin >> x;
		vx.push_back(i); // 对x坐标离散化!!!!
		event.push_back({perx[x], 0, i, 0});
		perx[x] = i;
	}
	for (int i = 1; i <= q; i++) {
		int x1, x2, y1, y2;
		cin >> x1 >> x2;
		y1 = 0, y2 = x1 - 1;
		event.push_back({y2, 2, x2, i}); // 2加
		event.push_back({y2, 1, x1 - 1, i}); // 1减
	}
	// 离散化 此时就是好的
	sort(event.begin(), event.end());
	m = n;
	//
	for (auto evt : event) {
		if (evt[1] == 0) {
			int x = upper_bound(vx.begin(), vx.end(), evt[2]) - vx.begin() + 1 - 1;
			change(x, 1);
		} else {
			int x = upper_bound(vx.begin(), vx.end(), evt[2]) - vx.begin() + 1 - 1;
			int tmp = query(x);
			if (evt[1] == 1)
				ans[evt[3]] -= tmp;
			else
				ans[evt[3]] += tmp;
//			if (evt[3] == 2 && evt[1] == 2) {
//				printf("__%d――――", bit[1]);
//			}
		}                    
//		  printf("y%d, op%d, x%d, rk%d==ans%d\n", evt[0], evt[1], evt[2], evt[3], ans[1]);
	}

	for (int i = 1; i <= q; i++) {
		cout << ans[i] << "\n";
	}

	return 0;
}
