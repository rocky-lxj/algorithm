#include<bits/stdc++.h>
using namespace std;
#define y1 abbbbb
using ll = long long;
const int N = 1e6 + 7;

// http://oj.daimayuan.top/course/15/problem/717
int n, cnt, a[N], l[N], r[N], ans[N];

void dfs(int u) {
	ans[u] = ++cnt;
	if (l[u])
		dfs(l[u]);
	if (r[u])
		dfs(r[u]);
}
void build() {
	stack<int> st; // 存的是下标
	int root = 0;

	for (int i = 1; i <= n; i++) {
		int last = 0;
		while (!st.empty() && a[i] < a[st.top()]) {
			last = st.top();
			st.pop();
		}
		if (!st.empty())
			r[st.top()] = i;
		else
			root = i;
		l[i] = last;
		st.push(i);
	}
	dfs(root);
}
int main() {
	cin >> n;
	for (int i = 1; i <= n; i++) {
		cin >> a[i];
	}
	//
	build();
	//
	for (int i = 1; i <= n; i++)
		cout << ans[i] << " ";
	return 0;
}
